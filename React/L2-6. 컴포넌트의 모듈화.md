# 모듈화의 필요성

- 컴포넌트의 복잡화로 인해 관리가 어려워지므로, 이를 관리하기 위해 분리하는 작업
- == 컴포넌트를 별도 파일로 정의하고 관리하는 것을 의미한다.

## 모듈화 전

+-------------------------------------------------------------+
| App.js (모든 컴포넌트와 로직) |
|-------------------------------------------------------------|
| import React, { useState } from 'react'; |
| |
| function App() { |
| const [count, setCount] = useState(0); |
| |
| // Header 컴포넌트 로직 |
| const appTitle = "My Awesome App"; |
| const AppHeader = () => ( |
| <header> |
| <h1>{appTitle}</h1> |
| </header> |
| ); |
| |
| // Button 컴포넌트 로직 |
| const AppButton = () => ( |
| <button onClick={() => setCount(count + 1)}> |
| Count: {count} |
| </button> |
| ); |
| |
| // Footer 컴포넌트 로직 |
| const AppFooter = () => ( |
| <footer> |
| <p>&copy; 2023 My Company</p> |
| </footer> |
| ); |
| |
| return ( |
| <div> |
| <AppHeader /> |
| <main> |
| <p>Welcome!</p> |
| <AppButton /> |
| </main> |
| <AppFooter /> |
| </div> |
| ); |
| } |
| |
| export default App; |
+-------------------------------------------------------------+

### 구조화

[ App.jsx ] <-- (전체 조립 및 레이아웃 담당)
│
┌─────┴─────┐
│ │
[ BackComp.jsx ] [ FrontComp.jsx ]
(데이터 통신 전담) (사용자 인터페이스 전담)

## App.jsx에서 import 하기

import FrontComp from "./FrontComp";
import FrontComp from "./BackComp";
을 상단에 적어 불러온 후
<FrontComp ...> 식으로 사용하면 된다.
